    def four_branch(x):
        x1 = x[0]
        x2  = x[1]
        k = x[2]
        g1 = 3+0.1*(x1-x2)**2-(x1+x2)/np.sqrt(2)
        g2 = 3+0.1*(x1-x2)**2+(x1+x2)/np.sqrt(2)
        g3 = (x1-x2)+k/np.sqrt(2)
        g4 =(x2-x1)+k/np.sqrt(2)
        return [min((g1,g2,g3,g4))]
